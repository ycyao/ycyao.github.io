02-线性结构3 Reversing Linked List

知识点总结：

vector容器中的begin(),end(),front(),与back()函数：

  1. iterator begin(),返回一个当前容器中起始元素的迭代器
  2. iterator end(),返回一个当前容器中末尾元素的迭代器
  3. reference front(),返回当前容器中起始元素的引用
  4. reference back(),返回当前容器中末尾元素的引用

代码如下：
思路：1.定义一个向量，下标为address值，根据节点的顺序，依次加入到新的vector中，对新的vector中的每k个元素进行反转，输出结果
使用空间换时间效率
2.实际上并没有完全意义上的对链表进行反转，输出时利用的节点间的关系完成输出。

#include<iostream>
#include<vector>
#include<iomanip>
#include<stdio.h>
using namespace std;
//
#define MAXA 100000
struct Node
{
    int address=-1;
    int data=0;
    int next=-1;
};

void reversenode(vector<Node>& list,int start,int end)
{
    while(start<end)
    {
        Node tmp=list[start];
        list[start]=list[end];
        list[end]=tmp;
        start++;
        end--;
    }
}

int main()
{
    int headadd,n,recount;
    cin>>headadd>>n>>recount;
    vector<Node> nodes(MAXA);

    for(int i=0;i<n;i++)
    {
        int addr,data,next;
        cin>>addr>>data>>next;
        nodes[addr].address=addr;
        nodes[addr].data=data;
        nodes[addr].next=next;
    }

    vector<Node> nodelist;
    nodelist.push_back(nodes[headadd]);
    while(nodelist.back().next!=-1)
    {
        nodelist.push_back(nodes[nodelist.back().next]);
    }

    int retime=nodelist.size()/recount;
    for(int i=0;i<retime;i++)
    {
        int start=i*recount;
        int end=start+recount-1;
        reversenode(nodelist,start,end);
    }
    /*for(int i=0;i<nodelist.size()-1;i++)
    {
        cout<<setfill('0')<<setw(5)<<nodelist[i].address<<" "<<nodelist[i].data<<" "<<setw(5)<<setfill('0')<<nodelist[i+1].address<<endl;
    }
    cout<<setfill('0')<<setw(5)<<nodelist.back().address<<" "<<nodelist.back().data<<" "<<"-1"<<endl;
    */
    for (int i = 0; i < nodelist.size()-1; i++)
    {
        printf("%05d %d %05d\n", nodelist[i].address, nodelist[i].data, nodelist[i+1].address);
    }
    printf("%05d %d %d\n", nodelist.back().address, nodelist.back().data, -1);

    return 0;

}
