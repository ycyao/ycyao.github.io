01-最大子序列和问题

描述：最大子序列和问题

使用在线处理算法：

#include<iostream>
using namespace std;

int main()
{
    int k,a[100000];
    cin>>k;
    for(int i=0;i<k;i++)cin>>a[i];
    int thissum=0,max=0;
    for(int i=0;i<k;i++)
    {
        thissum+=a[i];
        if(thissum>max)
            max=thissum;
        if(thissum<0)
            thissum=0;
    }
    cout<<max;
}
